//用无符号整型unsigned int 模拟32位寄存器 
uint32_t crc32(char * str, int len)
{
    uint32_t poly = 0x04C11DB7;  //多项式函数 
    uint32_t data = 0xffffffff; //初始化寄存器 
    int i = 0;
    for (i = 0; i < len; i++)
    {
        data ^= str[i]<<24; //由高字节向低字节循环处理 
        int j = 0;a
        for (j = 0; j < 8; j++)
        {
            if (data & 0x80000000) //最高位为1，就与多项式进行异或
                data = (data << 1) ^ poly;
            else
                data <<= 1;       
        }
    }
    return data ^ 0xffffffff; //反转运算 
}

uint32_t crc32(const char* str, int len) {
    uint32_t data = 0xffffffff; //初始化寄存器 
    int i = 0;
    for (i = 0; i < len; i++) {
        data = ((data & 0x00ffffff) << 8)  ^ crc32_table[(data >> 24) ^ (str[i]) & 0xff];
        //前面是保留处理后的字节 
        //后面由高位向低位处理，然后下标需要右移24位来处理 
    }
    return data ^ 0xffffffff;
}
